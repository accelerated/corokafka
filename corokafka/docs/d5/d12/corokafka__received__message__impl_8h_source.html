43 template <typename K, typename P>
44 ReceivedMessage<K,P>::~ReceivedMessage()
45 {
46  if (!_offsetSettings._autoOffsetPersist || _isPersisted) {
47  // auto-persistence is turned off or the offset has been persisted manually
48  return;
49  }
50  if (!_message) {
51  return;
52  }
53  if (std::uncaught_exception()) {
54  return; // don't commit if we are being destroyed as the result of an exception
55  }
56  if (_offsetSettings._autoOffsetPersistStrategy == OffsetPersistStrategy::Commit) {
57  if (_offsetSettings._autoCommitExec== ExecMode::Sync) {
58  _committer.commit(_message);
59  }
60  else { // async
61  _committer.get_consumer().async_commit(_message);
62  }
63  }
64  else {
65  _committer.get_consumer().store_offset(_message);
66  }
67  _isPersisted = true;
68 }

153 template <typename K, typename P>
154 void ReceivedMessage<K,P>::commit(void* opaque)
155 {
156  _opaque = opaque;
157  if (_committer.get_consumer().get_configuration().get_offset_commit_callback() && (_opaque != nullptr)) {
158  _offsets.insert(TopicPartition(getTopic(), getPartition(), getOffset()), opaque);
159  }
160  if (_offsetSettings._autoCommitExec == ExecMode::Sync) {
161  _committer.commit(_message);
162  }
163  else { // async
164  _committer.get_consumer().async_commit(_message);
165  }
166  _isPersisted = true;
167 }

174  _offsets.insert(TopicPartition(getTopic(), getPartition(), getOffset()), opaque);
175  }
176  _committer.get_consumer().store_offset(_message);
177  _isPersisted = true;
