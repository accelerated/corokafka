234 template <typename K, typename P>
235 size_t ProducerManagerImpl::send(const std::string& topic,
236  const K& key,
237  const P& payload,
238  const HeaderPack& headers,
239  void* opaque)
240 {
241  auto it = _producers.find(topic);
242  if (it == _producers.end()) {
243  throw std::runtime_error("Invalid topic");
244  }
245  ProducerTopicEntry& entry = it->second;
246  ConcreteMessageBuilder<ByteArray> builder = serializeMessage(entry, &key, &payload, &headers, opaque);
247  if (builder.topic().empty()) {
248  //Serializing failed
249  return 0;
250  }
251  builder.user_data(new PackedOpaque(opaque, std::promise<DeliveryReport>()));
252  if (!builder.payload().empty()) {
253  produceMessage(entry, builder);
254  }
255  return builder.payload().size();
256 }

258 template <typename K, typename P, typename HEADERS>
259 std::future<DeliveryReport> ProducerManagerImpl::postImpl(const std::string& topic,
260  K&& key,
261  P&& payload,
262  HEADERS&& headers,
263  void* opaque)
264 {
265  auto it = _producers.find(topic);
266  if (it == _producers.end()) {
267  throw std::runtime_error("Invalid topic");
268  }
269  ProducerTopicEntry& topicEntry = it->second;
270  if (topicEntry._payloadPolicy == Producer::PayloadPolicy::PASSTHROUGH_PAYLOAD) {
271  throw std::runtime_error("Invalid async operation for pass-through payload policy - use send() instead.");
272  }
273  if (topicEntry._preserveMessageOrder && (topicEntry._producer->get_buffer_size() > topicEntry._maxQueueLength)) {
274  throw std::runtime_error("Internal queue full");
275  }
276  std::promise<DeliveryReport> deliveryPromise;
277  std::future<DeliveryReport> deliveryFuture = deliveryPromise.get_future();
278 
279  // Post the serialization future and return
280  {
281  std::unique_lock<std::mutex> lock(_messageQueueMutex);
282  _messageQueue.emplace_back(_dispatcher.post<BuilderTuple>(
283  serializeCoro<K,P,HEADERS>,
284  it->second,
285  std::forward<K>(key),
286  std::forward<P>(payload),
287  std::forward<HEADERS>(headers),
288  new PackedOpaque(opaque, std::move(deliveryPromise))));
289  }
290  _emptyCondition.notify_one();
291  return deliveryFuture;
292 }
