27 class OffsetMap {

31  {}

33  // Indexed by partition and offset number
34  using MapType = std::map<TopicPartition, void*>;
35 
36  void insert(const TopicPartition& position, void* opaque) {
37  if (opaque == nullptr) return;
38  std::lock_guard<std::mutex> lock(*_mutex);
39  _map[position] = opaque;
40  }
41 
42  void insert(TopicPartition&& position, void* opaque) {
43  if (opaque == nullptr) return;
44  std::lock_guard<std::mutex> lock(*_mutex);
45  _map[std::move(position)] = opaque;
46  }
47 
48  void* find(const TopicPartition& position) const {
49  if (_map.empty()) return nullptr;
50  std::lock_guard<std::mutex> lock(*_mutex);
51  auto it = _map.find(position);
52  if (it == _map.end()) return nullptr;
53  return it->second;
54  }
55 
56  void* remove(const TopicPartition& position) {
57  if (_map.empty()) return nullptr;
58  std::lock_guard<std::mutex> lock(*_mutex);
59  auto it = _map.find(position);
60  if (it == _map.end()) return nullptr;
61  void* ptr = it->second;
62  _map.erase(it);
63  return ptr;
64  }
65 
66  void clear_by_partition(int partition) {
67  std::lock_guard<std::mutex> lock(*_mutex);
68  for (auto it = _map.begin(); it != _map.end();) {
69  if (it->first.get_partition() == partition) {
70  it = _map.erase(it);
71  }
72  else {
73  ++it;
74  }
75  }
76  }
77 
78  void clear() {
79  std::lock_guard<std::mutex> lock(*_mutex);
80  _map.clear();
81  }
82 
83  size_t size() const { return _map.size(); }
84 
85  bool empty() const { return _map.empty(); }
86 
87 private:
