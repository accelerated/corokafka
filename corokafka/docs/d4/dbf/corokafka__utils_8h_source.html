89 struct StringEqualCompare

92  {

94  return tolower (c1) == tolower (c2);

98 #if (__cplusplus == 201103L)
99  if (s1.size() == s2.size()) {
100  return std::equal(s1.begin(), s1.end(), s2.begin(), CharEqualCompare());

102  return false;
103 #else
104  return std::equal(s1.begin(), s1.end(), s2.begin(), s2.end(), CharEqualCompare());

108  // Compare the first 'n' characters
109  bool operator() (const std::string & s1, const std::string & s2, size_t num) const {
110  if ((s1.size() < num) || (s2.size() < num)) {
111  return false;
112  }
113 #if (__cplusplus == 201103L)

116  return std::equal(s1.begin(), s1.begin()+num, s2.begin(), s2.begin()+num, CharEqualCompare());
117 #endif

129 template<typename T, typename DT,

131 std::unique_ptr<T, DT> unique_pointer_cast(std::unique_ptr<U, DU>&& base, DT&& deleter = DT())
132 {
133  return std::unique_ptr<T, DT>(static_cast<T*>(base.release()), std::forward<DT>(deleter));
